c := 4e-9
setcellsize(c, c, c)
setsolver(2)
for e := 5; e < 14; e++ {
	n := pow(2, e)
	setgridsize(n, n, 1)
	print(n, "x", n)
	steps(1)
	b_ext = vector(0, 0.01, 0)
	m = uniform(1, 0, 0)
	steps(3)
	m = uniform(1, 0, 0)
	t = 0
	start := now()
	neval0 := Neval.get()
	steps(100)
	wall := since(start).Seconds()
	nevl := Neval.get() - neval0
	N2 := n * n
	fprintln("benchmark.txt", N2, N2*nevl/wall, t/nevl)
}
