Nxy := 32
Nz := 1
cellSize := 100e-9 / Nxy
SetGridSize(Nxy, Nxy, Nz)
SetCellSize(cellSize, cellSize, 5e-9/Nz)
Msat = 800e3
Aex = 13e-12
print("Exchange length:", Sqrt((2*Aex.GetRegion(0))/(mu0*Pow(Msat.GetRegion(0), 2))))
//Exchange length: 5.685802385313737e-09
TableAdd(E_total)
TableAdd(E_zeeman)
angle := 0.0
roundness := 1.0
size := 100
TableAddVar(angle, "Angle", "rad")
TableAddVar(roundness, "Roundness", "")
TableAddVar(size, "Size", "nm")
TableAddVar(cellSize, "Cell_size", "m")
angle0 := angle
roundness0 := roundness
size0 := size
for size = size0; size >= 100; size = size * 0.5 {
	for roundness = roundness0; (roundness >= 0.1) && (roundness*size > 128/Nxy); roundness -= 0.01 {
		print(sprintf("Short axis: %.2f nm", roundness*size))
		print(sprintf("Long axis: %d nm", size))
		geometry := Ellipse(size*1e-9, size*roundness*1e-9)
		geometry = geometry.Add(geometry.RotZ(Pi / 2))
		SetGeom(geometry)
		for angle = angle0; angle <= Pi/4; angle += Pi / 4 {
			m = Uniform(cos(angle), sin(angle), 0)
			B_ext = vector(0.001*cos(angle), 0.001*sin(angle), 0)
			minimize()
			TableSave()
		}
	}
}
