//Exchange length: 5.685802385313737e-09
TableAdd(E_total)
TableAdd(E_zeeman)
angle := 0.0
length := 90.
TableAddVar(angle, "Angle", "rad")
TableAddVar(length, "Length", "nm")
for length = 45.; length >= 42.; length -= 0.1 {
	print(sprintf("Short axis: %.1f nm", length))
	geometry := Ellipse(100e-9, length*1e-9)
	geometry = geometry.Add(geometry.RotZ(Pi / 2))
	SetGeom(geometry)
	for angle = 0.0; angle < 2*Pi; angle += 0.1 {
		m = Uniform(cos(angle), sin(angle), 0)
		B_ext = vector(5*cos(angle), 5*sin(angle), 0)
		minimize()
		TableSave()
	}
}
